\begin{enumerate}[label=\textbf{\underline{\arabic*.}}]
\item Скалярное произведение
	\begin{enumerate}
	      \item $\braket{\varphi}{\psi}$
            \item $\bra{\varphi}(\alpha_1\ket{\psi_1}+\alpha_2\ket{\psi_2})=\alpha_1\braket{\varphi}{\psi_1}+\alpha_2\braket{\varphi}{\psi_2}$
            \item $\braket{\psi}{\psi}\geq 0$
	\end{enumerate}
      $\braket{\psi}{\varphi}=\int\psi^*\varphi dx;\;\braket{\psi}{\psi}=\int\psi^*\psi dx =\int\abs{\psi}^2=\qnorm{\psi}\geq 0$
\item Коммутатор\\
      $\comm{A}{B}=\hat{A}\hat{B}-\hat{B}\hat{A}$
      \begin{enumerate}[label=\textbf{\underline{\alph*.}}]
            \item $ \comm{A}{B}=-\comm{B}{A} $
            \inlineitem $ \left[ \lambda\hat{A},\hat{B}\hat{C} \right] = \lambda\comm{A}{B} + \lambda\comm{A}{C} $
      \end{enumerate}
      $\comm{x}{p}=i\hbar$
\item $ \comml{A}{\hat{B}\hat{C}} = \hat{B}\comm{A}{C} + \comm{A}{B}\hat{C} $\\
      $ \commr{\hat{A}\hat{B}}{C} = \hat{A}\comm{B}{C} + \comm{A}{C}\hat{B} $
\item $ \hat{f}(\hat{A})=\sum\limits_{n=0}^{\infty}f_n \hat{A}^n $\\
      $\comml{A}{\hat{f}(\hat{A})} = 0\;\forall f $
\item $ e^{z\hat{A}}=\sum \frac{z^n}{n!}\hat{A}^n $ \\
      $ e^{z\hat{A}}e^{-z\hat{A}} = 1 $
\item $ \comm{A}{B}=\lambda \Rightarrow \comml{A}{\hat{f}(\hat{B})}=\lambda \pd{\hat{f}(\hat{B})}{\hat{B}} $
\item $ e^{z\hat{A}}\hat{B}e^{-z\hat{A}} = \hat{B} + \frac{z}{1!}\underbrace{\comm{A}{B}}_{K_1} + \frac{z^2}{2!}\underbrace{\comm{A}{K_1}}_{K_2} = \hat{B}+\sum\limits_{n=1}^{\infty}\frac{z^n}{n!}\comm{A}{K_n} $
\item Эрмитово сопряжение $\hat{A}\rightarrow\hat{A}^\dagger $\\
      $ \matrixel{\varphi}{\hat{A}}{\psi} = \matrixel{\psi}{\hat{A}^\dagger}{\varphi}^* $\\
      $ (\hat{A}^\dagger)^\dagger = \hat{A} $
\item $ (\hat{A_1} \cdot ... \cdot \hat{A_n})^\dagger = \hat{A_n}^\dagger \cdot ... \cdot \hat{A_1}^\dagger $
\item Самосопряженный оператор $\hat{A}^\dagger = \hat{A} $\\
      $ \matrixel{\varphi}{\hat{A}}{\psi} = \matrixel{\psi}{\hat{A}}{\varphi}^* $\\
      $ (e^{i\alpha\hat{A}})^\dagger = e^{-i\alpha\hat{A}} $
\item Унитарный оператор $ \hat{S}\hat{S}^\dagger = 1 $\\
      $ e^{i\alpha\hat{A}} $ - унитарен\\
      $ \hat{S}_a = e^\frac{i\hat{p}a}{\hbar}\; \Rightarrow\; \hat{x}' \rightarrow \hat{x}-a;\; \hat{p}'\rightarrow\hat{p}$\\
      $ \hat{S}_b = e^\frac{i\hat{x}b}{\hbar}\; \Rightarrow\; \hat{x}' \rightarrow \hat{x};\; \hat{p}'\rightarrow\hat{p}-b$\\
\item Момент импульса $\hat{L}=\hat{x}\times\hat{p};\;\hat{L}_i=\epsilon_{ijk}\hat{x}_j\hat{p}_k;\;\hat{l}_i=\frac1{\hbar}\hat{L}_i$\\
      $\left\{\begin{array}{l}
            \hat{l}_1=\frac1{\hbar}(\hat{x}_2\hat{p}_3-\hat{x}_3\hat{p}_2)\\
            \hat{l}_2=\frac1{\hbar}(\hat{x}_3\hat{p}_1-\hat{x}_1\hat{p}_3)\\
            \hat{l}_3=\frac1{\hbar}(\hat{x}_1\hat{p}_2-\hat{x}_2\hat{p}_1)
      \end{array}\right.$;
      $\left\{\begin{array}{l}
            \comm{l_i}{x_j}=i\epsilon_{ijk}\hat{x}_k\\
            \comm{l_i}{p_j}=i\epsilon_{ijk}\hat{p}_k\\
            \comm{l_i}{l_j}=i\epsilon_{ijk}\hat{l}_k
      \end{array}\right.$
\item Матрицы Паули\\
      $\hat{\sigma}_1=\begin{pmatrix} 0&1\\1&0 \end{pmatrix};\;
      \hat{\sigma}_2=\begin{pmatrix} 0&-i\\i&0 \end{pmatrix};\;
      \hat{\sigma}_3=\begin{pmatrix} 1&0\\0&-1 \end{pmatrix};\;
      \hat{E}=\begin{pmatrix} 1&0\\0&1 \end{pmatrix}$\\
      $\begin{pmatrix} a_{11} & a_{12} \\ a_{21} & a_{22} \end{pmatrix} =
      a_{11}\frac{\hat{E}+\hat{\sigma}_3}{2} +
      a_{22}\frac{\hat{E}-\hat{\sigma}_3}{2} +
      a_{12}\frac{\hat{\sigma}_1+i\hat{\sigma}_2}{2} +
      a_{21}\frac{\hat{\sigma}_1-i\hat{\sigma}_2}{2}$\\
      \begin{enumerate}[label=\textbf{\underline{\alph*.}}]
            \item $\hat{\sigma}_i^\dagger = \hat{\sigma}_i$
            \inlineitem $\hat{\sigma}_i^{-1} = \hat{\sigma}_i$
            \inlineitem $\hat{\sigma}_i\hat{\sigma}_j=-\hat{\sigma}_j\hat{\sigma}_i,\,i\neq j $
            \item $\Tr{\hat{\sigma}_i} = 0 ;\; \Tr{\hat{E}} = \Dim{\hat{E}}$
            \inlineitem $\hat{\sigma}_i\hat{\sigma}_j=\hat{E}\delta_{ij} + i\epsilon_{ijk}\hat{\sigma}_k$
      \end{enumerate}
\item $(\vec{a}\hat{\vec{\sigma}})(\vec{b}\hat{\vec{\sigma}}) = \hat{E}(\vec{a},\vec{b}) + i((\vec{a}\times\vec{b}),\hat{\vec{\sigma}})$
\item $ \abs{\vec{n}}=1;\; (\vec{n},\hat{\vec{\sigma}})^{2k} = \hat{E};\; (\vec{n},\hat{\vec{\sigma}})^{2k+1} = (\vec{n},\hat{\vec{\sigma}})$
\item $ e^{i\varphi(\vec{n},\hat{\vec{\sigma}})} = \hat{E}\cos{\varphi}+(\vec{n},\hat{\vec{\sigma}})\sin{\varphi} $
\item $\avg{\hat{A}}=\Tr{\hat{A}\hat{\rho}} $\\
      \begin{enumerate}[label=\textbf{\underline{\alph*.}}]
      \item $\Tr{\hat{\rho}}=1 $
      \inlineitem $\hat{\rho}^\dagger=\hat{\rho} $
      \inlineitem $\matrixel{f}{\hat{\rho}}{f}\ge0 $
      \end{enumerate}
\item $\hat{\rho}=\frac1{2}\hat{E}+\frac{r}{2}\scmult{\vec{n}}{\hat{\vec{\sigma}}} $
\item $\left\{\begin{array}{l}
            \hat{H}=\frac{\hat{p}^2}{2m}+\frac{m\omega^2}{2}\hat{x}^2\\
            \hat{H}\ket{\psi}=E\ket{\psi}\\
            \braket{\psi}{\psi}=1
      \end{array}\right.$
      $\left\{\begin{array}{l}
            \hat{x}=\frac{q_0}{\sqrt{2}}(\hat{a}+\hat{a}^\dagger),\;\;q_0=\sqrt{\frac{\hbar}{m\omega}}\\
            \hat{p}=\frac{p_0}{i\sqrt{2}}(\hat{a}-\hat{a}^\dagger),\;\;p_0=\sqrt{\hbar m\omega}\\
      \end{array}\right.\,\,q_0p_0=\hbar$\\
      $\left\{\begin{array}{l}
            \hat{a}=\frac1{\sqrt{2}}(\frac{\hat{x}}{q_0}+i\frac{\hat{p}}{p_0})\text{ - оператор уничтожения}\\
            \hat{a}^\dagger=\frac1{\sqrt{2}}(\frac{\hat{x}}{q_0}-i\frac{\hat{p}}{p_0})\text{ - оператор рождения}\\
      \end{array}\right.$\\
      $\comm{a}{a^\dagger}=1 $
\item $\hat{N}=\hat{a}^\dagger\hat{a};\;\;\hat{H}=\hbar\omega(\hat{N}+\frac1{2}) $\\
      $\left\{\begin{array}{l}
            \comm{N}{a}=-\hat{a}\\
            \comm{N}{a^\dagger}=\hat{a}^\dagger
      \end{array}\right.$
\item $\ket{0}\text{ - вакуумный вектор};\;\;\hat{N}\ket{0}=0\ket{0}$\\
      $\psi_0(x)=\frac1{4\sqrt{q_0^2\pi}}\cdot e^{-\frac{x^2}{2q_0^2}};\;\;
      \ket{\psi_n}=\ket{n}=\frac{(\hat{a}^\dagger)^n\ket{0}}{\sqrt{n!}} $
\item $\left\{\begin{array}{l}
            \hat{N}\ket{n}=n\ket{n}\\
            \braket{n}{n}=1
      \end{array}\right.$
      $\left\{\begin{array}{l}
            \hat{a}^\dagger=\sqrt{n-1}\ket{n+1}\\
            \hat{a}=\sqrt{n}\ket{n-1}
      \end{array}\right.$
\item Когерентные состояния\\
      $\left\{\begin{array}{l}
            \hat{a}\ket{\alpha}=\alpha\ket{\alpha}\\
            \braket{\alpha}{\alpha}=1
      \end{array}\right. $\\
      $\ket{\alpha}=\ket{\psi_\alpha}=\sum\limits_{n=0}^\infty\psi_n\ket{n}=
      \sum\limits_{n=0}^\infty\frac{\alpha^n}{\sqrt{n!}}\ket{n}\cdot e^{-\frac{\abs{\alpha}^2}{2}} $
\end{enumerate}
